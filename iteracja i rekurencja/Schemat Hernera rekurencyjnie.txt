#include <iostream>

using namespace std;

int horner(float n, int wsp[20],int x)
{
if (n==0) return wsp[0];                                                    //pozmienia≈Çem na inty, bo z floatami cos nie dzialalo xd
else return horner(n-1, wsp, x)*x+wsp[n]; // n to wielomian
}


int main()
{

int x, n, wsp[20];

cout<<"wielomian";
cin>>n; //wprowadzenie wielomiany
cout<<"Podaj x ";
cin>>x; //wsprwadzenie wspolczynnika
cout<<"wsp "<<endl;

for (int i=0;i<=n;i++)
cin>>wsp[i];
cout<<horner(n, wsp, x);
    return 0;
}
